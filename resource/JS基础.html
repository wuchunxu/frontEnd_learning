1

1  5  10 10 5  1	
*/
function yanghui(row){
	//第一行和第二行没有规律
	//从有规律开始，用循环
	var result = [[1],[1,1]];	
	for(var i=2;i < row;i++){
		result[i]=[1];//第i行的第一个值为1，也可以写成result.push([1]);
		for(var j = 1;j < i;j++){
			//第i行，从第二个值开始用循环，直到倒数第二个值;
			//第i行的第j个值等于第(i-1)行的j-1位和j位值之和
			result[i][j] = result[i-1][j-1] + result[i-1][j];
		}
		//第i行的最后一个值为1;
		//第一行有1个值，第二行有2个值，第i行有i个值。
		result[i][i] = 1;
	}
	return result;
}
